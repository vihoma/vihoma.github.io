#!/bin/sh

LC_ALL=fi_FI.UTF-8
MPD_HOST="olohuone.marko.homeunix.org";
export MPD_HOST;
STATUS=$(/opt/local/bin/mpc status|awk '/playing/{print $1}');

if [ -n $STATUS ] && [ "$STATUS" == "[playing]" ]; then
  echo "$(/opt/local/bin/mpc --format "[%artist%]" status|/usr/bin/head -1)";
else echo "";
fi
